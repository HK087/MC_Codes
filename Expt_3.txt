#include<reg51.h>
void Delay(unsigned int itime)
{	
	unsigned int i,j;
	for(i=0;i<itime;i++)
	{
		for(j=0;j<1275;j++);
	}
}
void main(void)
{
while(1)
	{	
	P1=0x00;
	Delay(15000);
	P1=0XFF;
	Delay(15000);
	}
}

#include<reg51.h>
void main(void)
{
	int z;
	while(1)
{	
	for(z=0;z<255;z++)
	{
		P1=z;
	}
	for(z=255;z>=0;z--)
	{
		P1=z;
	}
}
}
